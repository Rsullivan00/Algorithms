	int partition(Item A[], int left, int right)
	{
	  Item pivot = 0;	//We want 0 to be our pivot
	  int i = left;

	  for (int j = left; j <= right; ++j)	//Include final element
	    if (A[j] < pivot)
		swap(A[i++], A[j]);

	  //Omit final swap

	  return i;
